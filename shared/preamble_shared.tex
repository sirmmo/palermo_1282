%% — Font —
%% The *-Style filename shorthand (UprightFont=*-Regular, …) breaks for
%% families whose name contains a space when the on-disk files have no
%% space (EBGaramond-*.ttf, CinzelDecorative-*.ttf). Letting fontspec
%% auto-detect Bold/Italic/Bold Italic via fontconfig family lookup is
%% portable across local installs and CI.

%% — Stat block —
\newcommand{\statblock}[6]{%
  \begin{mdframed}[
    backgroundcolor=pergamena, linecolor=bordoscuro,
    linewidth=0.8pt, roundcorner=3pt,
    innerleftmargin=10pt, innerrightmargin=10pt,
    innertopmargin=8pt, innerbottommargin=8pt
  ]
  {\cinzel\normalsize\color{oro}\bfseries #1}
  \hfill
  {\cinzel\small\color{grigionote}\itshape #2}

  \begin{tabular*}{\linewidth}{@{}p{0.28\linewidth}p{0.14\linewidth}p{0.28\linewidth}p{0.14\linewidth}@{}}
    \textcolor{grigionote}{\small\PlaybookAttrA} & \textbf{#3} &
    \textcolor{grigionote}{\small\PlaybookAttrC} & \textbf{#5} \\[2pt]
    \textcolor{grigionote}{\small\PlaybookAttrB} & \textbf{#4} &
    \textcolor{grigionote}{\small\PlaybookAttrD} & \textbf{#6} \\
  \end{tabular*}
  \end{mdframed}%
}
